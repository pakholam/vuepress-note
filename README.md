  <CodeGroupItem title="YARN" active>

```bash
# Initialize environment
git init
yarn init

# install in your project
yarn add -D vuepress@next


# start writing
yarn docs:dev

# build to static files
yarn docs:uild
```

  </CodeGroupItem>

  <CodeGroupItem title="NPM">
  
```bash
# Initialize environment
git init
npx init

# install in your project
npm install -D vuepress@next

# start writing

npx  docs:dev

# build to static files

npx docs:build

```
